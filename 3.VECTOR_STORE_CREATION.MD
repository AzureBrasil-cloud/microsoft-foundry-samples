# criação da vector store

```bash
export AZURE_PROJECT_URL = 'a-url-do-seu-projeto'
```
```bash
curl -X POST "${AZURE_PROJECT_URL}/openai/vector_stores?api-version=2025-11-15-preview" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer ${AZURE_ACCESS_TOKEN}" \
  -d '{
    "name": "nome-da-sua-vectorstore",
    "expires_after": {
      "anchor": "last_active_at",
      "days": 365
    },
    "file_ids": ["FILE_ID"]
  }'
````
