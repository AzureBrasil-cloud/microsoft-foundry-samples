# attach da vector store no agente

```bash
AGENT_NAME="meu-agente-teste"
VECTOR_STORE_ID="vs_ID"
DEPLOYMENT_NAME="gpt-4.1-mini"
INSTRUCTIONS="You are a helpful assistant that answers questions about associated files only"

curl -X POST "${AZURE_PROJECT_URL}/agents/${AGENT_NAME}?api-version=2025-11-15-preview" \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer ${AZURE_ACCESS_TOKEN}" \
  -d "{
    \"definition\": {
      \"kind\": \"prompt\",
      \"model\": \"${DEPLOYMENT_NAME}\",
      \"instructions\": \"${INSTRUCTIONS}\",
      \"tools\": [
        {
          \"type\": \"file_search\",
          \"vector_store_ids\": [
            \"${VECTOR_STORE_ID}\"
          ]
        }
      ]
    }
  }"
```
